31bool MCHEmul::Instruction::matchesWith (const std::string& i, std::vector <std::string>& prms)
32{
33 if (_iStructure._error)
34 return (false);
35
36 std::string inst = _iStructure._templateWithNoParameters;
37
38 std::string iL = MCHEmul::upper (MCHEmul::noSpaces (i));
39
40 std::string pW = "";
41
42 // The one received can't be shorter than the template never
43 if (iL.length () < inst.length ())
44 return (false); // if it is, we are speaking about another opcode...
45
46 // If they don't start by the same symbol, thre woldn't be any possibility for them to match
47 if (iL [0] != inst [0])
48 return (false);
49
50 // It is time to review whether
51 // the structure of the received instruction could match with this...
52 prms = { };
53 bool result = true;
54 size_t ctIL = 0;
55 for (size_t ctInst = 0;
56 ctInst < inst.length () && ctIL < iL.length () && result; ctInst++, ctIL++)
57 {
58 // When the character in the received intruction
59 // doesn't match with the equivalent one in the template
60 // the system tries to find the next match, assuming that evrything
61 // In between should be a parameter...
62 // THis is something also checfed later!
63 if (iL [ctIL] != inst [ctInst])
64 {
65 size_t iPP = ctIL;
66 while (iL [ctIL] != inst [ctInst] && ctIL < iL.length ()) ctIL++;
67 if (result = (ctIL != iL.length ()))
68 {
69 prms.push_back (iL.substr (iPP, ctIL - iPP));
70
71 pW += '?';
72 pW += iL [ctIL];
73 }
74 }
75 else
76 pW += iL [ctIL];
77 }
78
79 // If the global aspect doesn't match
80 // It is not worth to continue...
81 if (!result)
82 return (false);
83
84 // If the instruction recevived hasn't been compared until its end,
85 // It could mean there is parameters still pending to add...
86 // That potential parameter should be a valid label or numbr!
87 if (ctIL != iL.length ())
88 {
89 std::string lP = iL.substr (ctIL);
90 if (MCHEmul::validLabel (lP) || MCHEmul::validBytes (lP))
91 {
92 prms.push_back (lP);
93 pW += '?';
94 }
95 else // In other case, it is not something that matches!
96 return (false);
97 }
98
99 // The number of extracted parameters has to match the number of the ones defined...
100 // ...and also the watermark!
101 return (_iStructure._waterMark == MCHEmul::onlyAlphanumeric (MCHEmul::removeAllFrom (iL, prms)) &&
102 _iStructure._waterMarkPlus == pW &&
103 prms.size () == _iStructure._parameters.size () );
104}
