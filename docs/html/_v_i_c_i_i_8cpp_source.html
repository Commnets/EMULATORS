148bool C64::VICII::simulate (MCHEmul::CPU* cpu)
149{
150 // Notice that the bad line detection routine takes into account
151 // the value of the YSCROLL register as the graphics information to be shown
152 // is loaded at the beginning of every bad line...
153 auto isBadRasterLine = [=]() -> bool
154 { return (!_VICIIRegisters -> videoResetActive () && _raster.isInPotentialBadLine () &&
155 ((_raster.currentLine () - 0x03) & 0x07 ) == _VICIIRegisters -> verticalScrollPosition ()); };
156
157 if (_VICIIRegisters -> vicIItoGenerateIRQ ())
158 cpu -> interrupt (F6500::IRQInterrupt::_ID) -> setActive (true);
159
160 // Rduce the visible zone if any... The info is passed to the raster!
161 _raster.reduceDisplayZone
162 (!_VICIIRegisters -> textDisplay25RowsActive (), !_VICIIRegisters -> textDisplay40ColumnsActive ());
163
164 for (size_t i = (cpu -> clockCycles () - _lastCPUCycles); i > 0 ; i--)
165 {
166 if (_isNewRasterLine)
167 {
168 if (isBadRasterLine () )
169 {
170 // This is not exactly what VICII does, but it could be a good aproximation...
171 // Not all graphic info is read actually at th beginning of a bad line, only video and color matrix...
172 // The rest of the info is read in further cycles depending on the type of graphic mode
173 readGraphicsInfo ();
174
175 cpu -> addClockCycles (_CPUCYCLESWHENREADGRAPHS); // The cost of reading in terms of cycles...
176 // This will slow down for a while later the cpu,
177 // that is what actually should happen!
178 }
179
180 if (_VICIIRegisters -> rasterIRQActive () &&
181 _raster.currentLine () == _VICIIRegisters -> IRQRasterLineAt ())
182 {
183 _VICIIRegisters -> setRasterAtLine (true);
184
185 cpu -> interrupt (F6500::IRQInterrupt::_ID) -> setActive (true);
186 }
187
188 _isNewRasterLine = false;
189 }
190
191 if (_raster.isInVisibleZone ())
192 {
193 // READ: Important variables
194 // Where is the raster in the visible part of the screen? (starting from 0)
195 unsigned short rv, cv; _raster.currentVisiblePosition (cv, rv);
196 // Which is the horizontal closest block of 8 pixels (from left to the right)?
197 // Take into account that the step - size of the raster is always 8 pixels per cycle..
198 // It always starts at 0...
199 unsigned short cav = (cv >> 3) << 3;
200
201 // Draws the border...
202 screenMemory () -> setHorizontalLine ((size_t) cav, (size_t) rv,
203 (cav + 8) > _raster.visibleColumns () ? (_raster.visibleColumns () - cav) : 8, _VICIIRegisters -> borderColor ());
204
205 if (_raster.isInDisplayZone () && // In the display...
206 rv >= _raster.vData ().firstScreenPosition () && // It has to be within the visible lines...
207 rv <= _raster.vData ().lastScreenPosition ( ) &&
208 !_VICIIRegisters -> videoResetActive ()) // ...and obviously with the video not reset!
209 {
210 // Draws the back ground, taken ito account
211 // The additional border that could exist...
212 unsigned short fpxl = 0; // First pixels from the cav to be drawn...
213 unsigned short lbk = 8; // Number of pixels to be drawn...
214 // With the screen?
215 if (cav <= _raster.hData ().lastScreenPosition ())
216 {
217 if (_raster.hData ().firstScreenPosition () > cav)
218 { fpxl += (_raster.hData ().firstScreenPosition () - cav); lbk = 8 - fpxl; }
219 if ((cav + 8) > _raster.hData ().lastScreenPosition ())
220 { lbk = _raster.hData ().lastScreenPosition () - cav + 1; }
221 screenMemory () -> setHorizontalLine ((size_t) (cav + fpxl), (size_t) rv, lbk, _VICIIRegisters -> backgroundColor ());
222 }
223
224 // Draw the graphics, and the sprites...
225 C64::VICII::DrawContext dC
226 = {
227 _raster.hData ().firstScreenPosition (true), // The original...
228 _raster.hData ().firstScreenPosition (), // And the real one
229 _raster.hData ().lastScreenPosition (true), // The original...
230 _raster.hData ().lastScreenPosition (), // And the real one
231 _VICIIRegisters -> horizontalScrollPosition (), // From 0 - 7
232 cv, cav, // Where the horizontal raster is (also adjusted)
233 _raster.vData ().firstScreenPosition (true), // The original...
234 _raster.vData ().firstScreenPosition (), // And the real one
235 _raster.vData ().lastScreenPosition (true), // The original...
236 _raster.vData ().lastScreenPosition (), // And the real one
237 _VICIIRegisters -> verticalScrollPosition (), // From 0 - 7 (taken into account in bad lines)
238 rv // Where the vertical raster is...
239 };
240
241 drawGraphics (dC);
242 drawSprites (dC);
243 }
244 }
245
246 // 1 cycle = 8 horizontal columns = 8 pixels...
247 _isNewRasterLine = _raster.moveCycles (1);
248
249 if (_raster.isInLastVBlank ())
250 {
251 if (!_lastVBlankEntered)
252 setGraphicsReady (_lastVBlankEntered = true); // The limit of the visible screen has been reached!
253 // so it is time to actualize the graphics...
254 }
255 else
256 _lastVBlankEntered = false;
257 }
258
259 // It might have been incremented after reading graphics...
260 _lastCPUCycles = cpu -> clockCycles ();
261
262 // To store back the info in the VIC Registers...
263 _VICIIRegisters -> setCurrentRasterLine (_raster.currentLine ());
264
265 unsigned short x1, y1, x2, y2;
266 _raster.screenPositions (x1, y1, x2, y2);
267 screenMemory () -> setHorizontalLine (x1 - 1, y1 - 1, x2 - x1 + 3, 1);
268 screenMemory () -> setHorizontalLine (x1 - 1, y2 + 1, x2 - x1 + 3, 1);
269 screenMemory () -> setVerticalLine (x1 - 1, y1 - 1, y2 - y1 + 3, 1);
270 screenMemory () -> setVerticalLine (x2 + 1, y1 - 1, y2 - y1 + 3, 1);
271
272 return (true);
273}

292void C64::VICII::drawGraphics (const C64::VICII::DrawContext& dC)
293{
294 // If no graphic has been loaded, it is not needed to continue...
295 if (_graphicsColorData.size () == 0)
296 return; // It could happen at the first lines of the screen when the vertical SCROLL is active...
297
298 // The graphical column being involved...
299 // The SCROLLX has been applied to the calculus, so it can be negative!
300 // -7 - _GRAPHMAXBITMAPCOLUMNS
301 int cb = (dC._RCA - dC._ICD) - dC._SC;
302
303 // The graphical line being involved...
304 // The SCROLLY is included in the calculus!
305 // The graphical data is loaded only in a bad line but "aligned" with the value of the SCROLLY
306 // So, at this point, the line being rastered could still have graphical data from the previous block of 8 rastered lines...
307 // 0 - _GRAPHMAXBITMAPROWS
308 size_t r = ((dC._RR - dC._IRS) % 8);
309 r = (r >= dC._SR) ? (r - dC._SR) : (8 + r - dC._SR);
310
311 switch (_VICIIRegisters -> graphicModeActive ())
312 {
313 case C64::VICIIRegisters::GraphicMode::_CHARMODE:
314 drawMonoColorBytes (cb, r, _graphicsCharData , _graphicsColorData, dC);
315 break;
316
317 case C64::VICIIRegisters::GraphicMode::_MULTICOLORCHARMODE:
318 drawMonoColorBytes (cb, r, _graphicsCharData, _graphicsColorData, dC);
319 break;
320
321 case C64::VICIIRegisters::GraphicMode::_BITMAPMODE:
322 drawMonoColorBytes (cb, r, _graphicsBitmapData , _graphicsColorData, dC);
323
324 case C64::VICIIRegisters::GraphicMode::_MULTICOLORBITMAPMODE:
325 drawMonoColorBytes (cb, r, _graphicsBitmapData, _graphicsColorData, dC);
326 break;
327
328 default:
329 // Not graphic mode supported, yet...
330 break;
331 }
332}

404void C64::VICII::drawMonoColorBytes (int cb, size_t r,
405 const MCHEmul::UBytes& bt, const MCHEmul::UBytes& clr, const C64::VICII::DrawContext& dC)
406{
407 for (int i = 0; i < 8 ; i++)
408 {
409 if ((cb + i) < 0)
410 continue;
411
412 size_t iBy = (size_t) ((cb + i) >> 3 );
413 size_t iBt = 7 - (size_t) ((cb + i) % 8);
414 unsigned short pos = dC._RCA + i;
415 if (pos <= dC._LCS && bt [(iBy << 3) + r].bit (iBt))
416 screenMemory () -> setPixel ((size_t) pos, (size_t) dC._RR,
417 (unsigned int) (clr [iBy].value () & 0x0f ));
418 }
419}

421// ---
422void C64::VICII::drawMultiColorBytes (int cb, size_t r,
423 const MCHEmul::UBytes& bt, const MCHEmul::UBytes& clr, const C64::VICII::DrawContext& dC)
424{
425 for (unsigned short i = 0 ; i < 8; i += 2)
426 {
427 if ((cb + i) < 0)
428 continue;
429
430 size_t iBy = (size_t) ((cb + i) >> 3);
431 size_t iBt = 7 - (size_t) ((cb + i) % 8);
432 unsigned char cs = (bt [(iBy << 3) + r].value () >> i) & 0x03; // 0, 1, 2 or 3...
433 unsigned int fc = (unsigned int) ((cs == 3)
434 ? clr [iBy].value () : _VICIIRegisters -> backgroundColor (cs)) & 0x0f ;
435 unsigned short pos = dC._RCA + i;
436 if (pos <= dC._LCS) screenMemory () -> setPixel ((size_t) pos, (size_t) dC._RR, fc);
437 if ((pos + 1) <= dC._LCS) screenMemory () -> setPixel ((size_t) (pos + 1), (size_t) dC._RR, fc);
438 }
