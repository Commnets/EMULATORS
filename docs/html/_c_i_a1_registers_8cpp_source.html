24void C64::CIA1Registers::setValue (size_t p, const MCHEmul::UByte& v)
25{
26 assert (_timerA != nullptr && _timerB != nullptr && _clock != nullptr);
27
28 MCHEmul::PhisicalStorageSubset::setValue (p, v);
29
30 size_t pp = p % 0x10;
31
32 switch (pp)
33 {
34 // Data Port Register A: CIAPRA
35 case 0x00:
36 // To indicate the row of the key matrix to be read (later)
37 // It will be affected by the communication direction (CIDDRA)
38 _keyboardRowToRead = v.value () & _dataPortADir;
39
40 // Data Port Register B: CIAPRB
41 case 0x01:
42 // No special meaning when set (but very important when read!)
43 break;
44
45 // Data Direction Register A: CIDDRA
46 case 0x02:
47 _dataPortADir = v.value ();
48 break;
49
50 // Data Direction Register B: CIDDRB
51 case 0x03:
52 _dataPortBDir = v.value ();
53 break;
54
55 // LSB of the Latch A: TIMALO
56 case 0x04:
57 _timerA -> setInitialValue (_timerA -> initialValue () & 0x00ff | (unsigned short) v.value ());
58 break;
59
60 // MSB of the Latch A: TIMAHI
61 case 0x05:
62 _timerA -> setInitialValue (_timerA -> initialValue () & 0xff00 | (unsigned short) (v.value () << 8));
63 break;
64
65 // LSB of the Latch B: TIMBLO
66 case 0x06:
67 _timerB -> setInitialValue (_timerB -> initialValue () & 0x00ff | (unsigned short) v.value ());
68 break;
69
70 // MSB of the Latch B: TIMBHI
71 case 0x07:
72 _timerB -> setInitialValue (_timerB -> initialValue () & 0xff00 | (unsigned short) (v.value () << 8));
73 break;
74
75 // Time of Day Clock Tenths of Seconds: TO2TEN
76 // Bits 0-3: BCD Digits. Bits 4-7: Unused.
77 case 0x08:
78 {
79 int ts = MCHEmul::UInt ({ MCHEmul::PhisicalStorageSubset::readValue (0x08) },
80 true , true ).asInt (); // A BCD value to int...
81 if (MCHEmul::PhisicalStorageSubset::readValue (0x0f).bit (7)) _clock -> setAlarmTenthSeconds ((unsigned char) ts);
82 else _clock -> setTenthSeconds ((unsigned char) ts);
83 }
84
85 break;
86
87 // Time of Day Clock Seconds: TO2SEC
88 // Bits 0-3: Second BCD Digit. Bits 4-6: First BCD Digit. Bit 7: Unused.
89 case 0x09:
90 {
91 int s = MCHEmul::UInt ({ MCHEmul::PhisicalStorageSubset::readValue (0x08) },
92 true , true ).asInt (); // A BCD value to int...
93 if (MCHEmul::PhisicalStorageSubset::readValue (0x0f).bit (7)) _clock -> setSeconds ((unsigned char) s);
94 else _clock -> setSeconds ((unsigned char) s);
95 }
96
97 break;
98
99 // Time of Day Clock Minutes: TO2MIN
100 // Bits 0-3: Second BCD Digit. Bits 4-6: First BCD Digit. Bit 7: Unused.
101 case 0x0b:
102 {
103 int m = MCHEmul::UInt ({ MCHEmul::PhisicalStorageSubset::readValue (0x08) },
104 true , true ).asInt (); // A BCD value to int...
105 if (MCHEmul::PhisicalStorageSubset::readValue (0x0f).bit (7)) _clock -> setAlarmMinutes ((unsigned char) m);
106 else _clock -> setMinutes ((unsigned char) m);
107 }
108
109 break;
110
111 // Time of Day Clock Hours: TO2HRS
112 // Bits 0-3: Second BCD Digit. Bit 4: First BCD Digit. Bits 5-6: Unused. Bit 7: AM/PM Flag (PM = 1)
113 case 0x0a:
114 {
115 int h = MCHEmul::UInt ({ MCHEmul::PhisicalStorageSubset::readValue (0x08) },
116 true , true ).asInt (); // A BCD value to int...
117 if (MCHEmul::PhisicalStorageSubset::readValue (0x0f).bit (7)) _clock -> setAlarmHours ((unsigned char) h);
118 else _clock -> setHours ((unsigned char) h);
119 }
120
121 break;
122
123 // Serial Data Port CI2SDR
124 case 0x0c:
125 break;
126
127 // Interrupt Control Register: CI2ICR
128 // Depending on the bit 7 the behaviour is different: 1 = bits with 1 are set, 0 = bits with 1 are cleared...
129 case 0x0d:
130 if (v.bit (0)) _timerA -> setIRQEnabled (v.bit (7));
131 if (v.bit (1)) _timerB -> setIRQEnabled (v.bit (7));
132 if (v.bit (2)) _clock -> setIRQEnabled (v.bit (7));
133 // TODO: To manage the serial port...
134 break;
135
136 // Control Register A: CI2CRA
137 case 0x0e:
138 _timerA -> setEnabled (v.bit (0));
139 // TODO: Sending signals to serial port...(bits 1, 2 & 6)
140 _timerA -> setRunMode (v.bit (3) ? CIATimer::RunMode::_ONETIME : CIATimer::RunMode::_RESTART);
141 if (v.bit (4)) _timerA -> reset ();
142 _timerA -> setCountMode (v.bit (5)
143 ? C64::CIATimer::CountMode::_PROCESSORCYCLES: C64::CIATimer::CountMode::_SIGNALSONCNTLINE);
144 // TODO: Pending to control based on 60Hz or 50 Hz...(bit 7)
145 break;
146
147 // Control Register B: CI2CRB
148 case 0x0f:
149 _timerB -> setEnabled (v.bit (0));
150 // TODO: Sending signals to serial port B...(bits 1, 2 & 6)
151 _timerB -> setRunMode (v.bit (3) ? CIATimer::RunMode::_ONETIME : CIATimer::RunMode::_RESTART);
152 if (v.bit (4)) _timerB -> reset ();
153 // bits 5 & 6 indicates the mode...
154 _timerB -> setCountMode ((C64::CIATimer::CountMode) ((v.value () >> 4) & 0x03));
155 // _PROCESSORCYCLES = 0, _SIGNALSONCNTLINE = 1,...
156 break;
157
158 default:
159 break;
160 }

164const MCHEmul::UByte& C64::CIA1Registers::readValue (size_t p) const
165{
166 assert (_timerA != nullptr && _timerB != nullptr && _clock != nullptr);
167
168 MCHEmul::UByte result = MCHEmul::PhisicalStorage::_DEFAULTVALUE;
169
170 size_t pp = p % 0x10;
171
172 switch (pp)
173 {
174 case 0x00:
175 result = MCHEmul::UByte (_joystick2Status);
176 break;
177
178 case 0x01:
179 {
180 // Identify the row to read...
181 // Take into account that the row is identified writting 0 in its bit (so it is necessary to inverse it first)
182 unsigned char bRV = ~_keyboardRowToRead;
183 size_t c = 0; while ((bRV >>= 1) != 0) c++;
184 // If more than 1 row will be possible, the greatest one will be choosen...
185 // The columns (from keyboard or joystick 1) connected will have its bits to 0!...
186 result = _keyboardStatusMatrix [c] & ~_dataPortBDir;
187 }
188
189 break;
190
191 // When reading no special behaviour...but when setting!
192 case 0x02:
193 case 0x03:
194 result = MCHEmul::PhisicalStorageSubset::readValue (p);
195 break;
196
197 case 0x04:
198 result = MCHEmul::UByte ((unsigned char) (_timerA -> currentValue () & 0x00ff));
199 break;
200
201 case 0x05:
202 result = MCHEmul::UByte ((unsigned char) ((_timerA -> currentValue () & 0xff00) >> 8));
203 break;
204
205 case 0x06:
206 result = MCHEmul::UByte ((unsigned char) (_timerB -> currentValue () & 0x00ff));
207 break;
208
209 case 0x07:
210 result = MCHEmul::UByte ((unsigned char) ((_timerB -> currentValue () & 0xff00) >> 8));
211 break;
212
213 case 0x08:
214 // Only 1 digit from 0 to 9 (4 bits). Rest unused.
215 result = MCHEmul::UInt::fromUnsignedInt (_clock -> tenthsSecond (), MCHEmul::UInt::_PACKAGEDBCD).bytes ()[0];
216 break;
217
218 case 0x09:
219 // 2 digits. 1º: from 0 to 9 (4 bits), 2º: from 0 to 6 (3 bits). Bit 7 unused
220 result = MCHEmul::UInt::fromUnsignedInt (_clock -> seconds (), MCHEmul::UInt::_PACKAGEDBCD).bytes ()[0];
221 break;
222
223 case 0x0a:
224 // 2 digits. 1º: from 0 to 9 (4 bits), 2º: from 0 to 6 (3 bits). Bit 7 unused
225 result = MCHEmul::UInt::fromUnsignedInt (_clock -> minutes (), MCHEmul::UInt::_PACKAGEDBCD).bytes ()[0];
226 break;
227
228 case 0x0b:
229 {
230 // 2 digits. 1º: from 0 to 9 (4 bits), 2º: either 0 or 1 (1 bit). Bits 5,6 unused. Bit 7 AM/PM flag (PM = 1)
231 unsigned int h = (unsigned int) _clock -> hours ();
232 bool pm = h > 12 || (h == 12 && _clock -> minutes () > 0);
233 if (pm && h > 12) h -= 12; // The value 12 is kept...
234 result = MCHEmul::UInt::fromUnsignedInt (h, MCHEmul::UInt::_PACKAGEDBCD).bytes ()[0];
235 result.setBit (7, pm);
236 }
237
238 break;
239
240 case 0x0c:
241 // TODO:
242 // To simulate the Serial Port.
243 // The directions of the information has to be taken into account
244 // When the serial port is configurated for input, the data is read in the pint 5 when the pin 4 (CNT) let you know
245 // the info is ready for reading (this is connected with the timer).
246 break;
247
248 case 0x0d:
249 {
250 bool IA = _timerA -> IRQRequested (); // It is set back to false after reading...
251 bool IB = _timerB -> IRQRequested (); // Same...
252 bool IC = _clock -> IRQRequested (); // Same...
253 result = MCHEmul::UByte::_0;
254 result.setBit (7, IA || IB || IC); // IRQ?
255 result.setBit (0, IA); // in Timer A?
256 result.setBit (1, IB); // in Timer B?
257 result.setBit (2, IC); // in Clock?
258 // TODO: To manage the serial port...
259 }
260
261 break;
262
263 case 0x0e:
264 case 0x0f:
265 result = MCHEmul::PhisicalStorageSubset::readValue (p);
266 break;
267
268 default:
269 break;
270 }
271
272 return (_lastValueRead = result);
273}
