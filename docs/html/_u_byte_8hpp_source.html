156 inline UByte UByte::bitAdding (const UByte& u, bool cin, bool& cout, bool& o) const
157 {
158 unsigned short r =
159 (unsigned short) (_value + u._value +
160 (cin ? 0x01 : 0x00));
161
162 // The carry is generated if the result is bigger than a unsigned char long!
163 cout = (r & 0xff00 ) != 0x000;
164 // The overflow is generated when both are negative and the outcome positive...or the other way around!
165 o = (bit (7) && u.bit (7) && ((r & 0x0080) == 0x0000)) |
166 (!bit (7) && !u.bit (7) && ((r & 0x0080) != 0x0000));

168 return (MCHEmul::UByte ((unsigned char) r));
