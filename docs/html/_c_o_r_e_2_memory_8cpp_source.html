39bool MCHEmul::PhisicalStorageSubset::load (const std::string& fN, size_t sA, bool bE)
40{
41 std::ifstream file (fN, std::ios::binary);
42 if (!file)
43 return (false);
44
45 file.seekg (0, std::ios::end);
46 std::streamoff lF = file.tellg ();
47 if (lF > (std::streamoff) std::numeric_limits <size_t>::max () ||
48 lF < (std::streamoff) sA || (lF - (std::streamoff) sA) > (std::streamoff) size ())
49 return (false); // either bad format or too long for this memory...
50
51 size_t lFA = (size_t) lF;
52 char* aDT = new char [sA];
53 file.seekg (0, std::ios::beg);
54 file.read (aDT, (std::streamsize) sA); // Reads the address where to load the info
55 char* fDT = new char [lFA - sA];
56 file.read (fDT, (std::streamsize) (lFA - sA)); // Reads the info itself
57
58 file.close ();
59
60 std::vector <MCHEmul::UByte> aV;
61 for (size_t i = 0; i < (size_t) (sA / MCHEmul::UByte::size ()); i += MCHEmul::UByte::size ())
62 aV.push_back ((MCHEmul::UByte) (*(aDT + i)));
63 std::vector <MCHEmul::UByte> fV;
64 for (size_t i = 0; i < (size_t) ((lFA - sA) / MCHEmul::UByte::size ()); i += MCHEmul::UByte::size ())
65 fV.push_back ((MCHEmul::UByte) (*(fDT + i)));
66
67 delete [] aDT;
68 delete [] fDT;
69
70 int dt;
71 MCHEmul::Address adr (aV, bE);
72 if (!isIn (adr, dt))
73 return (false);
74
75 set (adr, fV, true);
